How to use

Creating new Modals is very simple.
Go ahead, open up your console and try some of these.



Modal.alert('Alert Title', 'This is an alert message.');


Modal.alert('Alert Title', 'This is an alert message.', function(){
	// this is the callback for when the "ok" button is clicked
});


Modal.alert('Alert Title Only', '');


Modal.alert('This is an alert with message only, no title');


// show a spinner
var spinner = Modal.spinner();

// close all modals
Modal.close();



// those methods above are just shortcuts to creating Modals.
// You can call the modal method on its own for complete control

new Modal({
	title: 'Title',
	msg: 'The message text here. This could be HTML, or a jQuery object',
	effect: 1, // 2, 3, 4, 5, 6 ... see top of Modal.js for style names
	btns: [
		'close', // shortcut to create close button
		{label: 'Another Button', className: 'button-primary', onClick: function(){}};
	]
})
